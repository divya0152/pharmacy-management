create database mydata;

use mydata;

CREATE TABLE pharmacy(
REFRENCENO INTEGER NOT NULL PRIMARY KEY,
MEDICINETYPE VARCHAR(50) NOT NULL,
MEDICINENAME VARCHAR(50) NOT NULL,
LOTNO INTEGER,
ISSUEDATE DATE,
EXPDATE DATE,
DOSAGE VARCHAR(50),
PRICE VARCHAR(50),
PRODUCTQT VARCHAR(50)
);
